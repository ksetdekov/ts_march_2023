def get_seasonal_period(ts):
    """  

    Args:
        ts:  

    Returns:
        int:  ,   
    """    
    f, ppx = periodogram(ts)
    top_feq = list(sorted(zip(f, ppx), key=lambda x: x[1]))[-1]
    period = np.round(1/top_feq[0], 0)
    return period
